    ```typescript
    import { broadcastToRoom } from '../websocket/rooms';
    import type { OrderWithItems } from '../repositories/order';
    
    export interface NotificationPayload {
      type: string;
      timestamp: string;
      data: unknown;
    }
    
    // Check if WebSocket is available (skip if not initialized yet)
    function safeBroadcast(room: string, payload: NotificationPayload) {
      try {
        broadcastToRoom(room, JSON.stringify(payload));
      } catch (err) {
        // WebSocket not ready, log but don't fail
        console.log(`[Notification] Would send to ${room}:`, payload.type);
      }
    }
    
    export function emitOrderCreated(order: OrderWithItems) {
      const payload = {
        type: 'order:created',
        timestamp: new Date().toISOString(),
        data: {
          orderId: order.id,
          tableNumber: order.tableNumber,
          items: order.items?.map(item => ({
            name: item.menuName,
            quantity: item.quantity,
            notes: item.notes,
            status: item.status
          })) || [],
          total: order.total,
          createdAt: order.createdAt
        }
      };
      
      // Notify kitchen of new order
      safeBroadcast('kitchen', payload);
      // Also notify admin dashboard
      safeBroadcast('admin', payload);
    }
    
    export function emitOrderStatusChanged(
      orderId: number, 
      status: string, 
      kitchenStatus?: string,
      tableNumber?: number
    ) {
      const payload = {
        type: 'order:status-changed',
        timestamp: new Date().toISOString(),
        data: {
          orderId,
          status,
          kitchenStatus,
          tableNumber
        }
      };
      
      // Status changes affect both kitchen and cashier
      safeBroadcast('kitchen', payload);
      safeBroadcast('cashier', payload);
      safeBroadcast('admin', payload);
    }
    
    export function emitOrderCompleted(orderId: number, tableNumber?: number) {
      const payload = {
        type: 'order:completed',
        timestamp: new Date().toISOString(),
        data: {
          orderId,
          tableNumber,
          completedAt: new Date().toISOString()
        }
      };
      
      // Completion notification primarily for cashier
      safeBroadcast('cashier', payload);
      safeBroadcast('admin', payload);
    }
    
    export function emitPaymentReceived(
      orderId: number, 
      amountPaid: number, 
      method: string,
      tableNumber?: number
    ) {
      const payload = {
        type: 'payment:received',
        timestamp: new Date().toISOString(),
        data: {
          orderId,
          amountPaid,
          method,
          tableNumber,
          receivedAt: new Date().toISOString()
        }
      };
      
      // Payment confirmation for cashier
      safeBroadcast('cashier', payload);
      safeBroadcast('admin', payload);
    }
    
    // Export a singleton-like interface
    export const notifications = {
      emitOrderCreated,
      emitOrderStatusChanged,
      emitOrderCompleted,
      emitPaymentReceived
    };
    ```
    
    Key design decisions:
    - Safe broadcast: Don't fail if WebSocket not ready (allows startup ordering flexibility)
    - Targeted rooms: Kitchen gets order-created, both get status-changed
    - Full context: Include table number, items, totals for complete information
