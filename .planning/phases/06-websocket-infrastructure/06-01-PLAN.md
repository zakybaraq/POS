    ```javascript
    class POSWebSocket {
      constructor(options = {}) {
        this.url = options.url || `ws://${window.location.host}/ws`;
        this.token = options.token;
        this.reconnectInterval = options.reconnectInterval || 3000;
        this.maxReconnects = options.maxReconnects || 10;
        this.listeners = new Map();
        this.reconnectCount = 0;
        this.shouldReconnect = true;
      }
      
      connect() {
        const wsUrl = `${this.url}?token=${encodeURIComponent(this.token)}`;
        this.ws = new WebSocket(wsUrl);
        
        this.ws.onopen = () => {
          console.log('WebSocket connected');
          this.reconnectCount = 0;
          this.emit('connected');
        };
        
        this.ws.onmessage = (event) => {
          const data = JSON.parse(event.data);
          this.emit(data.type, data.payload);
        };
        
        this.ws.onclose = () => {
          this.emit('disconnected');
          if (this.shouldReconnect && this.reconnectCount < this.maxReconnects) {
            setTimeout(() => this.connect(), this.reconnectInterval);
            this.reconnectCount++;
          }
        };
        
        this.ws.onerror = (error) => {
          this.emit('error', error);
        };
      }
      
      on(event, handler) {
        if (!this.listeners.has(event)) {
          this.listeners.set(event, []);
        }
        this.listeners.get(event).push(handler);
      }
      
      emit(event, data) {
        const handlers = this.listeners.get(event) || [];
        handlers.forEach(h => h(data));
      }
      
      send(type, payload) {
        if (this.ws?.readyState === WebSocket.OPEN) {
          this.ws.send(JSON.stringify({ type, payload }));
        }
      }
      
      disconnect() {
        this.shouldReconnect = false;
        this.ws?.close();
      }
    }
    
    // Export for module systems
    if (typeof module !== 'undefined') {
      module.exports = { POSWebSocket };
    }
    ```
    
    This client handles:
    - Automatic reconnection with exponential backoff
    - Event-based message handling
    - Token-based authentication via query param
    - Connection state management

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP → WebSocket | JWT token crosses from HTTP cookie/auth to WebSocket connection |
| Client → Server | Untrusted client input via WebSocket messages |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01 | Spoofing | WebSocket auth | mitigate | Validate JWT on every connection; reject without valid token |
| T-06-02 | Tampering | Message payload | accept | Messages are notifications only; no state-changing operations via WS |
| T-06-03 | Repudiation | Connection logs | mitigate | Log all connections/disconnections with userId and timestamp |
| T-06-04 | DoS | Connection flooding | mitigate | idleTimeout: 120s, maxPayloadLength: 1MB; rate limit connection attempts |
| T-06-05 | Elevation | Room access | mitigate | Join rooms based on JWT role; never trust client room requests |
</threat_model>
